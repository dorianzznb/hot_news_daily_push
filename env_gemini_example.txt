# Gemini 配置示例 - 使用 Google Gemini 作为总结模型

# =========================
# 总结模型配置
# =========================
SUMMARY_MODEL=gemini

# =========================
# Google Gemini API 配置
# =========================
GEMINI_API_KEY=your_gemini_api_key_here
GEMINI_BASE_URL=https://gemini.kbz.ink
GEMINI_MODEL_NAME=gemini-2.0-flash-exp

# =========================
# 腾讯混元 API 配置 (用于单条摘要生成)
# =========================
HUNYUAN_API_KEY=your_hunyuan_api_key_here

# =========================
# 推送配置 (至少配置一种)
# =========================
WEBHOOK_URL=your_webhook_url_here

# 或者配置具体的推送渠道
# 企业微信机器人
# QYWX_KEY=your_qywx_key_here

# 钉钉机器人
# DD_BOT_SECRET=your_dd_bot_secret_here
# DD_BOT_TOKEN=your_dd_bot_token_here

# =========================
# 错误通知配置 (可选，独立的错误推送通道)
# =========================
# 启用/禁用错误通知功能
ERROR_NOTIFICATION_ENABLED=true

# 专门用于错误通知的企业微信机器人 (推荐)
# ERROR_QYWX_KEY=your_error_qywx_key_here

# 专门用于错误通知的Telegram机器人
# ERROR_TG_BOT_TOKEN=your_error_tg_bot_token_here
# ERROR_TG_USER_ID=your_telegram_user_id

# 专门用于错误通知的Webhook URL
# ERROR_WEBHOOK_URL=your_error_webhook_url_here

# =========================
# 运行模式配置
# =========================
TECH_ONLY=false
SKIP_CONTENT=false
NO_CACHE=false

# =========================
# 使用说明
# =========================
# 1. 获取 Gemini API 密钥（通过代理服务提供方）
# 2. 将 your_gemini_api_key_here 替换为真实的API密钥
# 3. 复制此文件为 .env
# 4. 运行测试: python tests/test_gemini_integration.py

# =========================
# 测试 Gemini 连接的命令
# =========================
# export GEMINI_API_KEY=your_actual_api_key
# export GEMINI_BASE_URL=https://gemini.kbz.ink
# python tests/test_gemini_integration.py

# =========================
# 常见错误解决
# =========================
# 400 Bad Request (API_KEY_INVALID): API密钥无效，请检查密钥是否正确
# 400 Bad Request (User location is not supported): 地理位置限制，请使用VPN
# 401 Unauthorized: 认证失败，请检查API密钥权限
# 403 Forbidden: API密钥权限不足或未激活
# 40058 (微信webhook): markdown内容超长(>4096字符)，已自动优化控制

# =========================
# 错误通知说明
# =========================
# 新增的错误通知功能将在系统出现异常时通过独立渠道发送通知，避免干扰正常推送：
# 1. 配置错误 - 发送错误通知后程序退出
# 2. 网络连接失败 - 发送错误通知后程序退出
# 3. 数据收集失败 - 发送错误通知后程序退出
# 4. 内容处理失败 - 发送错误通知但程序继续（使用原始内容）
# 5. AI总结失败 - 发送错误通知后程序退出
# 6. 推送失败 - 发送错误通知后程序退出
# 
# 测试错误通知: python tests/test_error_notification.py

# 注意：代理服务使用 x-goog-api-key 认证头
# 注意：Google Gemini API 在某些地区受限，如遇地理位置错误请使用VPN
# 注意：已优化内容长度控制，每个热点最多关联3个新闻源
# 注意：错误通知渠道应与正常推送渠道分离，避免用户被错误信息打扰
